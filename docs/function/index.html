    public compareVersion(versionA: Array, operator: String, versionB: Array): boolean since 1.0.0 source 

    import {compareVersion} from '@wetransfer/concorde/modules/browser/src/version.js'

    Compares to semantic versions, given an operator.

    Params:

    NameTypeAttributeDescription
    versionA Array

    One of the versions to compare.
    operator String

    Compare versions using this operator. We support >, >=, <, <=, = and ==.
    versionB Array

    The other version to compare.

    Return:

    boolean

    True if condition can be satisfied, false otherwise.

    Example:

    
compareVersion('1.0.1', '<', '5.1.0')
// => true

compareVersion('4.4.1', '=', '0.1.0')
// => false

compareVersion('4.4.1', '>=', '6.1.0')
// => false

    public diffVersions(versionA: Array, versionB: Array): number since 1.0.0 source 

    import {diffVersions} from '@wetransfer/concorde/modules/browser/src/version.js'

    Compares to versions (array format) and determines the difference between them.

    Params:

    NameTypeAttributeDescription
    versionA Array

    One of the versions to compare.
    versionB Array

    The other version to compare.

    Return:

    number

    Returns the difference.

    Example:

    
diffVersions([1, 0, 0], [1, 5, 0])
// => -5

diffVersions([1, 0, 0], [1, 0, 0])
// => 0

diffVersions([3, 5, 1], [1, 1, 0])
// => 2

    import {versionToArray} from '@wetransfer/concorde/modules/browser/src/version.js'

    Converts a string version into an array

    Params:

    NameTypeAttributeDescription
    version String

    The version to convert, in semantic versioning format.

    Return:

    Array

    Returns the version in array format.

    Example:

    
versionToArray('1.0.1')
// => [1, 0, 1]

versionToArray('1.0.1-alpha')
// => [1, 0, 1]

versionToArray([3, 5, 1], [1, 1, 0])
// => 2
